## Получение картинки с сайта

Получение картинки по проекту с сервера
```
	СтруктураURI = ОбщегоНазначенияКлиентСервер.СтруктураURI(АдресWebПриложения);
	Схема = ?(ЗначениеЗаполнено(СтруктураURI.Схема), СтруктураURI.Схема, "http");
	Прокси = ПолучениеФайловИзИнтернетаКлиентСервер.ПолучитьПрокси(Схема);
	
	
	Соединение = Новый HTTPСоединение(
	СтруктураURI.Хост,
	СтруктураURI.Порт,
	, //СтруктураURI.Логин,
	,  //СтруктураURI.Пароль, 
	Прокси,
	60,
	?(НРег(Схема) = "http", Неопределено, Новый ЗащищенноеСоединениеOpenSSL),Истина);
	
	HttpЗапрос = Новый HTTPЗапрос(СтруктураURI.ПутьНаСервере);
	HttpЗапрос.АдресРесурса =СтруктураURI.ПутьНаСервере;
	
	Результат = Соединение.Получить(HttpЗапрос);
	
	ДанныеКартинки = Результат.ПолучитьТелоКакДвоичныеДанные();

```
